342 bool OMAxis::moveMotor(bool dir, uint32_t steps,
343  uint32_t totalTravelTime, uint32_t accelTime, uint32_t decelTime) {
344  OMByteBuffer sendBuf;
345  uint8_t _buf[22];
346  sendBuf.assign(_buf, sizeof(_buf));
347  sendBuf.append((uint8_t) CMD_PC_MOVE_MOTOR_CPLX);
348  sendBuf.append(dir);
349  sendBuf.append(htonl(steps));
350  sendBuf.append((uint8_t)OM_PDU_ESC);
351  sendBuf.append(htonl(totalTravelTime));
352  sendBuf.append((uint8_t)OM_PDU_ESC);
353  sendBuf.append(htonl(accelTime));
354  sendBuf.append((uint8_t)OM_PDU_ESC);
355  sendBuf.append(htonl(decelTime));
356  int res = command(m_slaveAddr, OM_PCODE_PC, sendBuf.getBuf(),
357  sendBuf.size());
358  if (res == 0)
359  return true;
360  else
361  return false;
362 }
