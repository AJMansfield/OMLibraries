138 LRESULT QextSerialEnumerator::onDeviceChangeWin( WPARAM wParam, LPARAM lParam )
139 {
140  if ( DBT_DEVICEARRIVAL == wParam || DBT_DEVICEREMOVECOMPLETE == wParam )

143  if( pHdr->dbch_devicetype == DBT_DEVTYP_DEVICEINTERFACE )

150  }
151  }
152  return 0;

155 bool QextSerialEnumerator::matchAndDispatchChangedDevice(const QString & deviceID, const GUID & guid, WPARAM wParam)
156 {
157  bool rv = false;
158  DWORD dwFlag = (DBT_DEVICEARRIVAL == wParam) ? DIGCF_PRESENT : DIGCF_ALLCLASSES;
159  HDEVINFO devInfo;
160  if( (devInfo = SetupDiGetClassDevs(&guid,NULL,NULL,dwFlag)) != INVALID_HANDLE_VALUE )
161  {
162  SP_DEVINFO_DATA spDevInfoData;
163  spDevInfoData.cbSize = sizeof(SP_DEVINFO_DATA);
164  for(int i=0; SetupDiEnumDeviceInfo(devInfo, i, &spDevInfoData); i++)
165  {
166  DWORD nSize=0 ;
167  TCHAR buf[MAX_PATH];
168  if ( SetupDiGetDeviceInstanceId(devInfo, &spDevInfoData, buf, MAX_PATH, &nSize) &&
169  deviceID.contains(TCHARToQString(buf))) // we found a match
170  {
171  rv = true;
172  QextPortInfo info;
173  info.productID = info.vendorID = 0;
174  getDeviceDetailsWin( &info, devInfo, &spDevInfoData, wParam );
175  if( wParam == DBT_DEVICEARRIVAL )
176  emit deviceDiscovered(info);
177  else if( wParam == DBT_DEVICEREMOVECOMPLETE )
178  emit deviceRemoved(info);
179  break;
180  }
181  }
182  SetupDiDestroyDeviceInfoList(devInfo);
183  }
184  return rv;
185 }

187 bool QextSerialEnumerator::getDeviceDetailsWin( QextPortInfo* portInfo, HDEVINFO devInfo, PSP_DEVINFO_DATA devData, WPARAM wParam )
188 {
189  portInfo->friendName = getDeviceProperty(devInfo, devData, SPDRP_FRIENDLYNAME);
190  if( wParam == DBT_DEVICEARRIVAL)
191  portInfo->physName = getDeviceProperty(devInfo, devData, SPDRP_PHYSICAL_DEVICE_OBJECT_NAME);
192  portInfo->enumName = getDeviceProperty(devInfo, devData, SPDRP_ENUMERATOR_NAME);
193  QString hardwareIDs = getDeviceProperty(devInfo, devData, SPDRP_HARDWAREID);
194  HKEY devKey = SetupDiOpenDevRegKey(devInfo, devData, DICS_FLAG_GLOBAL, 0, DIREG_DEV, KEY_READ);
195  portInfo->portName = QextSerialPort::fullPortNameWin( getRegKeyValue(devKey, TEXT("PortName")) );
196  QRegExp idRx("VID_(\\w+)&PID_(\\w+)");
197  if( hardwareIDs.toUpper().contains(idRx) )
198  {
199  bool dummy;
200  portInfo->vendorID = idRx.cap(1).toInt(&dummy, 16);
201  portInfo->productID = idRx.cap(2).toInt(&dummy, 16);
202  //qDebug() << "got vid:" << vid << "pid:" << pid;
203  }
204  return true;
205 }
