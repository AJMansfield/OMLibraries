145 uint8_t OMMoCoBus::getPacket() {
146 
147  uint8_t command_val = 0;
148  uint8_t com_byte_count = 0;
149  //uint8_t dat = 0;
150  uint8_t ret = 0;
151 
152  bool not_us = false;
153 
154  // we check serial.available() here even
155  // though this->_getNextByte() does, because
156  // a timeout is applied waiting for next command
157  // in this->_getNextByte() - since this is called
158  // every cycle of the main loop, we don't want
159  // to introduce a delay unless we know of a
160  // byte waiting to be read
161 
162  int avail = m_serObj->available();
163 
164  // serial does not have data in the buffer
165  if( avail <= 0 )
166  return(0);
167 
168  // is this packet intended for us?
169  uint8_t stat = this->_targetUs();
170 
171  if(stat == OM_SER_ERR || stat == OM_SER_TIMEOUT) {
172  // got an error? timeout error, really.
173  // this->_flushSerial();
174  return(0);
175  }
176  else if( stat != OM_SER_OK ) {
177  // command was for someone else
178  // (but we need to read the rest of the data
179  // off the line to make sure we don't keep part of it
180  // in the buffer)
181  not_us = true;
182  }
183 
184  // get command byte
185  ret = this->_getNextByte(command_val);
186 
187  if ( ret != OM_SER_OK ) {
188  // well, this is just embarassing
189  // this->_flushSerial();
190  return(0);
191  }
192 
193  if( command_val > 0 ) {
194  // get data length (even if not our own command)
195  ret = this->_getNextByte(com_byte_count);
196 
197  if ( ret != OM_SER_OK ) {
198  // well, this is just embarassing
199  // this->_flushSerial();
200  return(0);
201  }
202  }
203 
204  // check for overflow
205  // note: we will still be in error condition, as
206  // there will be values left in the serial buffer
207  // on the next read, from this command - so don't
208  // exceed it, eh? This is just to prevent memory
209  // corruption. It is not safe otherwise
210 
211  if(com_byte_count > OM_SER_BUFLEN)
212  com_byte_count = OM_SER_BUFLEN;
213 
214 
215  // clear out any previous command data
216  memset(m_serBuffer, 0, sizeof(uint8_t) * OM_SER_BUFLEN);
217  m_bufSize = 0;
218 
219 
220  // populate command data buffer
221  for( int i = 1; i <= com_byte_count; i++ ) {
222  // get com_byte_count character values from the serial
223  // buffer
224 
225  uint8_t dat = 0;
226  uint8_t ret = this->_getNextByte(dat);
227 
228  if( ret != OM_SER_OK )
229  return(0);
230 
231  m_serBuffer[i - 1] = dat;
232  m_bufSize++;
233  }
234 
235  if( not_us == true ) {
236  // the command was not intended for us
237  memset(m_serBuffer, 0, sizeof(uint8_t) * OM_SER_BUFLEN);
238  return(0);
239  }
240 
241 
242  return( command_val );
243 }

446 uint8_t OMMoCoBus::_targetUs() {
447 
448  uint8_t thsChar = 1;
449  uint8_t ret = this->_getNextByte(thsChar);
450 
451  // timeout? other error?
452  if( ret != OM_SER_OK )
453  return(OM_SER_TIMEOUT);
454 
455  // if character is null...
456  if( thsChar == 0 ) {
457  // until we get enough nulls for a start command
458  // or read something that isn't null...
459 
460  uint8_t nullCount = 0;
461 
462  while( nullCount < 5 && thsChar == 0 ) {
463  nullCount++;
464 
465  // null count met
466  if( nullCount == 5 )
467  break;
468 
469  ret = this->_getNextByte(thsChar);
470 
471  if( ret != OM_SER_OK )
472  return(OM_SER_TIMEOUT);
473 
474  }
475 
476  // did not receive a proper start sequence
477  if( nullCount < 5 )
478  return(OM_SER_ERR);
479 
480  thsChar = 0;
481 
482  ret = this->_getNextByte(thsChar);
483 
484  if( ret != OM_SER_OK )
485  return(OM_SER_TIMEOUT);
486 
487  if( thsChar != 255 )
488  return(OM_SER_ERR);
489 
490  // get target address
491 
492  unsigned int addr = 0;
493  ret = this->_getNextByte(thsChar);
494 
495  if( ret != OM_SER_OK )
496  return(OM_SER_TIMEOUT);
497 
498  addr = (unsigned int) thsChar << 8;
499 
500  ret = this->_getNextByte(thsChar);
501 
502  if( ret != OM_SER_OK )
503  return(OM_SER_TIMEOUT);
504 
505  addr |= (unsigned int) thsChar;
506 
507  // is not destined for us?
508  if( addr != m_devAddr )
509  return(OM_SER_NOT_US);
510 
511  } // end if(thsChar == 0
512  else {
513  // uint8_t invalid
514  return(OM_SER_ERR);
515  }
516 
517  return(OM_SER_OK);
518 
519 }

521 uint8_t OMMoCoBus::_getNextByte(uint8_t& p_data) {
522 
523  uint8_t ret = OM_SER_OK;
524  uint8_t* p = (uint8_t*)(void*) &p_data;
525 
526  if( m_serObj->available() > 0 ) {
527  *p = (uint8_t) m_serObj->read();
528  }
529  else {
530  unsigned long timer = Utils::millis();
531  // wait until we have data ready
532  while( m_serObj->available() < 1 ) {
533  if( Utils::millis() - timer > OM_SER_WAIT ) {
534  // timed out waiting for complete input
535  return( OM_SER_TIMEOUT );
536  }
537  }
538  *p = (uint8_t) m_serObj->read();
539  }
540 
541 
542  return(ret);
543 }
