239 qint64 QextSerialPort::writeData(const char *data, qint64 maxSize)
240 {
241  QMutexLocker lock( mutex );
242  DWORD retVal = 0;
243  if (queryMode() == QextSerialPort::EventDriven) {
244  OVERLAPPED* newOverlapWrite = new OVERLAPPED;
245  ZeroMemory(newOverlapWrite, sizeof(OVERLAPPED));
246  newOverlapWrite->hEvent = CreateEvent(NULL, true, false, NULL);
247  if (WriteFile(Win_Handle, (void*)data, (DWORD)maxSize, & retVal, newOverlapWrite)) {
248  CloseHandle(newOverlapWrite->hEvent);
249  delete newOverlapWrite;
250  }
251  else if (GetLastError() == ERROR_IO_PENDING) {
252  // writing asynchronously...not an error
253  QWriteLocker writelocker(bytesToWriteLock);
254  _bytesToWrite += maxSize;
255  pendingWrites.append(newOverlapWrite);
256  }
257  else {
258  qDebug() << "serialport write error:" << GetLastError();
259  lastErr = E_WRITE_FAILED;
260  retVal = (DWORD)-1;
261  if(!CancelIo(newOverlapWrite->hEvent))
262  qDebug() << "serialport: couldn't cancel IO";
263  if(!CloseHandle(newOverlapWrite->hEvent))
264  qDebug() << "serialport: couldn't close OVERLAPPED handle";
265  delete newOverlapWrite;
266  }
267  } else if (!WriteFile(Win_Handle, (void*)data, (DWORD)maxSize, & retVal, NULL)) {
268  lastErr = E_WRITE_FAILED;
269  retVal = (DWORD)-1;
270  }
271  return (qint64)retVal;
272 }

791 /*
792  Triggered when there's activity on our HANDLE.
793 */
794 void QextSerialPort::onWinEvent(HANDLE h)
795 {
796  QMutexLocker lock(mutex);
797  if(h == overlap.hEvent) {
798  if (eventMask & EV_RXCHAR) {
799  if (sender() != this && bytesAvailable() > 0)
800  emit readyRead();
801  }
802  if (eventMask & EV_TXEMPTY) {
803  /*
804  A write completed. Run through the list of OVERLAPPED writes, and if
805  they completed successfully, take them off the list and delete them.
806  Otherwise, leave them on there so they can finish.
807  */
808  qint64 totalBytesWritten = 0;
809  QList<OVERLAPPED*> overlapsToDelete;
810  foreach(OVERLAPPED* o, pendingWrites) {
811  DWORD numBytes = 0;
812  if (GetOverlappedResult(Win_Handle, o, & numBytes, false)) {
813  overlapsToDelete.append(o);
814  totalBytesWritten += numBytes;
815  } else if( GetLastError() != ERROR_IO_INCOMPLETE ) {
816  overlapsToDelete.append(o);
817  qWarning() << "CommEvent overlapped write error:" << GetLastError();
818  }
819  }
820 
821  if (sender() != this && totalBytesWritten > 0) {
822  QWriteLocker writelocker(bytesToWriteLock);
823  emit bytesWritten(totalBytesWritten);
824  _bytesToWrite = 0;
825  }
826 
827  foreach(OVERLAPPED* o, overlapsToDelete) {
828  OVERLAPPED *toDelete = pendingWrites.takeAt(pendingWrites.indexOf(o));
829  CloseHandle(toDelete->hEvent);
830  delete toDelete;
831  }
832  }
833  if (eventMask & EV_DSR) {
834  if (lineStatus() & LS_DSR)
835  emit dsrChanged(true);
836  else
837  emit dsrChanged(false);
838  }
839  }
840  WaitCommEvent(Win_Handle, &eventMask, &overlap);
841 }
