305 bool OMAxis::moveMotor(bool dir, uint32_t steps,
306  uint32_t totalTravelTime, uint32_t accelTime, uint32_t decelTime) {
307  ByteBuffer sendBuf;
308  uint8_t _buf[22];
309  sendBuf.assign(_buf, sizeof(_buf));
310  sendBuf.append((uint8_t) CMD_PC_MOVE_MOTOR_CPLX);
311  sendBuf.append(dir);
312  sendBuf.append(htonl(steps));
313  sendBuf.append((uint8_t)OM_PDU_ESC);
314  sendBuf.append(htonl(totalTravelTime));
315  sendBuf.append((uint8_t)OM_PDU_ESC);
316  sendBuf.append(htonl(accelTime));
317  sendBuf.append((uint8_t)OM_PDU_ESC);
318  sendBuf.append(htonl(decelTime));
319  int res = command(slave_addr, OM_PCODE_PC, sendBuf.getBuf(),
320  sendBuf.size());
321  if (res == 0)
322  return true;
323  else
324  return false;
325 }
